def thick_font(image):
    import numpy as np
    image = cv2.bitwise_not(image) # invert image
    kernel = np.ones((2, 2), np.uint8)
    image = cv2.dilate(image, kernel, iterations=1)
    image = cv2.bitwise_not(image)
    return (image)
